// vertex shader
#version 300 es

layout (location = 0) in vec3 aPos;
layout (location = 1) in vec2 tPos;
out vec2 TexCoord;

void main()
{
    gl_Position = vec4(aPos, 1.0);
    TexCoord = tPos;
}


// fragment shader
#version 300 es
precision mediump float;

out vec4 FragColor;
in vec2 TexCoord;
uniform sampler2D texture1;

void main()
{
    FragColor = texture (texture1, TexCoord);
}
